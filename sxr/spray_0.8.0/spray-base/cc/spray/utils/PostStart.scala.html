package cc.spray.utils

import akka.actor.Actor

/**
 * Mixin for actors.
 * Provides a `postStart` callback.
 */
trait PostStart {
  this: Actor =>

  def postStart()

  override def preStart() {
    become {
      case 'PostStart => {
        try {
          postStart()
        } finally {
          unbecome()
        }
      }
    }
    self ! 'PostStart
  }
}
