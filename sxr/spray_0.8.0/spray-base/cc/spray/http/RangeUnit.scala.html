package cc.spray.http

sealed abstract class RangeUnit {
  def value: String
  override def toString = value
}

object RangeUnits {

  val bytes = new RangeUnit {
    def value = "bytes"
  }

  case class CustomRangeUnit(value: String) extends RangeUnit
}
